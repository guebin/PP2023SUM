def grade(attendance, report, mid, final): 
    if attendance<70: 
        credit = 'F' 
    else: 
        total_score = attendance * 0.1 + report * 0.2 + mid * 0.3 + final * 0.4 
        if total_score > 80: 
            credit = 'A+' 
        else:

    return credit
